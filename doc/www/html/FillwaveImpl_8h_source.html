573 inline void Engine::EngineImpl::drawScene(GLfloat time) {
574 
575  evaluateTime(time);
576 
577  if (mPostProcessingPasses.size()) {
578  auto _compare_function = [](flc::PostProcessingPass &pass) -> bool {
579  return pass.isFinished();
580  };
581  auto _begin = mPostProcessingPasses.begin();
582  auto _end = mPostProcessingPasses.end();
583 
584  flc::Texture2DRenderableDynamic *textureNext;
585  flc::Texture2DRenderableDynamic *textureCurrent = (*_begin).getFrame();
586 
587  flc::Program *programCurrent;
588 
589  drawClear();
590  textureCurrent->bindForWriting();
591  drawSceneCore();
592 
593  for (auto it = _begin; it != _end; it++) {
594 
595  auto next = it + 1;
596 
597  if (next != _end) {
598 
599  textureNext = (*next).getFrame();
600  textureCurrent = (*it).getFrame();
601  programCurrent = (*it).getProgram();
602 
603  textureNext->bindForWriting();
604 
605  drawClear();
606 
607  textureCurrent->draw(0.0f);
608 
609  textureNext->draw(time);
610 
611  textureNext->bindForReading();
612 
613  } else {
614 
615  flc::Framebuffer::bindScreenFramebuffer();
616 
617  // render to current bound framebuffer using textureCurrent as a texture to post process
618  textureCurrent->draw(time);
619 
620  textureCurrent = (*it).getFrame();
621  programCurrent = (*it).getProgram();
622 
623  // render to current bound framebuffer using textureCurrent as a texture to post process
624  drawTexture(textureCurrent, programCurrent);
625  }
626 
627  (*it).checkTime(time);
628  }
629 
630  auto it = remove_if(_begin, _end, _compare_function);
631 
632  mPostProcessingPasses.erase(it, _end);
633 
634  } else {
635  drawSceneCore();
636  }
637 }

682 inline void Engine::EngineImpl::evaluateShadowMaps() {
683 
684  glDepthMask(GL_TRUE);
685 
686  mLights->updateLightEntities();
687 
688  GLint currentTextureUnit = FILLWAVE_SHADOW_FIRST_UNIT;
689 
690  flc::Texture2DRenderable *light2DTexture;
691  for (size_t i = 0; i < mLights->mLightsSpot.size(); i++) {
692  light2DTexture = mLights->mLightsSpot[i]->getShadowTexture();
693  light2DTexture->bindForWriting();
694  light2DTexture->bind(currentTextureUnit++);
695  glClearColor(0.0, 0.0, 0.0, 1.0);
696  glClear(GL_DEPTH_BUFFER_BIT);
697  mScene->drawDepth(*(mLights->mLightsSpot[i]->getShadowCamera()));
698  }
699 
700  for (size_t i = 0; i < mLights->mLightsDirectional.size(); i++) {
701  light2DTexture = mLights->mLightsDirectional[i]->getShadowTexture();
702  light2DTexture->bindForWriting();
703  light2DTexture->bind(currentTextureUnit++);
704  glClearColor(0.0, 0.0, 0.0, 1.0);
705  glClear(GL_DEPTH_BUFFER_BIT);
706  mScene->drawDepth(*(mLights->mLightsDirectional[i]->getShadowCamera()));
707  }
708 
709  for (size_t i = 0; i < mLights->mLightsPoint.size(); i++) {
710  flf::LightPoint *lightPoint = mLights->mLightsPoint[i].get();
711  flc::Texture3DRenderable *light3DTexture = lightPoint->getShadowTexture();
712  glm::vec3 lightPosition(lightPoint->getTranslation());
713  light3DTexture->bindForWriting();
714  light3DTexture->bind(currentTextureUnit);
715  for (GLint j = GL_TEXTURE_CUBE_MAP_POSITIVE_X; j <= GL_TEXTURE_CUBE_MAP_NEGATIVE_Z; j++) {
716  light3DTexture->setAttachmentFace(j, GL_COLOR_ATTACHMENT0);
717  glClearColor(0.0, 0.0, 0.0, 1.0);
718  glClear(GL_DEPTH_BUFFER_BIT);
719  mScene->drawDepthColor(*(lightPoint->getShadowCamera(j)), lightPosition);
720  }
721  currentTextureUnit++;
722  }
723  flc::Framebuffer::bindScreenFramebuffer();
724 }

736 inline void Engine::EngineImpl::evaluateDebugger() {
737  GLint mCurentTextureUnit = 0;
738  GLint id = 0;
739  switch (mDebugger->getState()) {
740  case eDebuggerState::eLightsSpot:
