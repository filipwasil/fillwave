69  template <class T>
70  static void handleEvent(std::vector<T> &callbacks, EventType &event) {
71  /* Run callbacks */
72  for (auto &it : callbacks) {
73  if (it->isEnabled()) {
74  if (it->getEventType() == event.getType()) {
75  it->perform(event);
76  }
77  }
78  }

80  /* Erase finished callbacks */
81  auto _find_finished_function = [](T &m) -> bool {
82  return m->isFinished();
83  };
84  auto _begin = callbacks.begin();
85  auto _end = callbacks.end();
86  auto it = std::remove_if(_begin, _end, _find_finished_function);
87  callbacks.erase(it, _end);
88  }
