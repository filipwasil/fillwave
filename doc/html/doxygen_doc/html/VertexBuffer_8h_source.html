68  void getAttributes(GLint programHandle) {
69  int howMany = -1;
70  glGetProgramiv(programHandle, GL_ACTIVE_ATTRIBUTES, &howMany);
71  for (int i = 0; i < howMany; ++i) {
72  int name_len = -1, num = -1;
73  GLenum type = GL_ZERO;
74  char name[200];
75  glGetActiveAttrib(programHandle, GLuint(i), sizeof(name) - 1,
76  &name_len, &num, &type, name);
77  name[name_len] = 0;
78  GLuint location = glGetAttribLocation(programHandle, name);
79  GLuint size = 0;
80  unsigned long long pointer = 0;
81  switch (type) {
82  case GL_UNSIGNED_INT:
83  case GL_FLOAT:
84  case GL_INT:
85  case GL_BOOL:
86  size = 1;
87  break;
88  case GL_UNSIGNED_INT_VEC2:
89  case GL_FLOAT_VEC2:
90  case GL_INT_VEC2:
91  case GL_BOOL_VEC2:
92  size = 2;
93  break;
94  case GL_UNSIGNED_INT_VEC3:
95  case GL_FLOAT_VEC3:
96  case GL_INT_VEC3:
97  case GL_BOOL_VEC3:
98  size = 3;
99  break;
100  case GL_UNSIGNED_INT_VEC4:
101  case GL_FLOAT_VEC4:
102  case GL_INT_VEC4:
103  case GL_BOOL_VEC4:
104  case GL_FLOAT_MAT2:
105  size = 4;
106  break;
107  case GL_FLOAT_MAT3:
108  size = 9;
109  break;
110  case GL_FLOAT_MAT4:
111  size = 16;
112  break;
113  default:
114  std::cout << "Not supported type of attribute" << std::endl;
115  size = 0;
116  break;
117  }
118  Attribute a(name, location, size, sizeof(T), type);
119  mAttributes.push_back(a);
120  }
121 
122  unsigned long long pointer = 0;
123  for (GLuint i = 0; i < mAttributes.size(); i++) {
124  for (auto& it : mAttributes) {
125  if (it.getIndex() == i) {
126  it.setOffsetPointer((GLvoid*) pointer);
127  pointer += it.getSize() * it.getTypeSize();
128  }
129  }
130  }
131  }
